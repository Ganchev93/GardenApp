# GardenCare Features Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Мигрирай всички PoC pages към Firebase, добави responsive layout (desktop sidebar + header + footer), FCM push notifications, AI visual diary, freemium gates и admin panel. Deploy в production.

**Architecture:** Foundation Plan е завършен — имаме Auth, Firestore, Cloudinary, Firebase Functions с Gemini. Тук мигрираме UI-то от localStorage към Firebase hooks, изграждаме responsive layout с Tailwind `md:` breakpoints (mobile = bottom nav, desktop = sidebar + header + footer), добавяме FCM за push notifications и Admin panel достъпен само при `role === 'admin'`.

**Tech Stack:** React 19, Vite, Tailwind CSS v4, React Router v7, Firebase v10, Vitest, NavLink за active states

## Global Constraints

- Всички breakpoints: mobile-first — default стилове за mobile, `md:` за desktop (≥768px)
- Sidebar само на `md:` и нагоре, BottomNav само под `md:`
- Header е sticky top-0 на всички размери
- Footer само на desktop (`hidden md:block`)
- `useGarden(uid)` идва от Foundation Plan — не дублирай localStorage логика
- Freemium limits: free = 3 растения + 20 AI анализа/месец; premium = unlimited; admin = unlimited
- Admin panel достъпен само при `profile.role === 'admin'` — проверка на frontend И backend
- FCM: поискай permission само след user interaction, не при зареждане
- Commit след всяка завършена задача

```jsx
import { BrowserRouter, Routes, Route } from 'react-router-dom'

import Auth from './pages/Auth'
import Home from './pages/Home'
import Plants from './pages/Plants'
```

```jsx
import { useState } from 'react'
import { plants as catalog } from '../data/plants'
import { useAuth } from '../hooks/useAuth'
import { useGarden } from '../hooks/useGarden'
import { useFreemium } from '../hooks/useFreemium'
import FreemiumGate from '../components/ui/FreemiumGate'

function toDateStr(ts) {
  if (!ts) return ''
  const d = ts.toDate ? ts.toDate() : new Date(ts)
  return d.toISOString().slice(0, 10)
}

function addDays(dateStr, days) {
  const d = new Date(dateStr)
  d.setDate(d.getDate() + days)
  return d.toISOString().slice(0, 10)
}

export default function MyGarden() {
  const { user } = useAuth()
  const { plants, loading, addPlant, markWatered, markFertilized, updateNote, removePlant } = useGarden(user?.uid)
  const { canAddPlant } = useFreemium()
  const [showAdd, setShowAdd] = useState(false)
  const [selectedId, setSelectedId] = useState('')
  const [lastWatered, setLastWatered] = useState(new Date().toISOString().slice(0, 10))
  const [lastFertilized, setLastFertilized] = useState(new Date().toISOString().slice(0, 10))
  const [activeTab, setActiveTab] = useState('water')
  const [photoView, setPhotoView] = useState(null)

  const today = new Date().toISOString().slice(0, 10)

  const waterDue = plants.filter(p => toDateStr(p.nextWatering) <= today)
  const waterUpcoming = plants.filter(p => toDateStr(p.nextWatering) > today)
  const fertilizeDue = plants.filter(p => toDateStr(p.nextFertilizing) <= today)
  const fertilizeUpcoming = plants.filter(p => toDateStr(p.nextFertilizing) > today)

  async function handleAdd() {
    if (!selectedId) return
    const plant = catalog.find(p => p.id === Number(selectedId))
    if (!plant) return
    await addPlant({
      plantId: plant.id,
      name: plant.name,
      emoji: plant.emoji,
      category: plant.category,
      photo: null,
      watering_frequency_days: plant.watering.frequency_days,
      watering_amount: plant.watering.amount,
      frequency_days: plant.fertilizing.frequency_days,
      fertilizer_type: plant.fertilizing.fertilizer_type,
      dose: plant.fertilizing.dose,
    })
    setShowAdd(false)
    setSelectedId('')
  }

  if (loading) return <div className="text-center py-12 text-gray-400">Зареждане...</div>

  return (
    <div>
      {photoView && (
        <div className="fixed inset-0 bg-black/80 z-50 flex items-center justify-center p-4" onClick={() => setPhotoView(null)}>
          <img src={photoView} alt="plant" className="max-w-full max-h-full rounded-2xl object-contain" />
        </div>
      )}

      <div className="flex items-center justify-between mb-4">
        <div>
          <h1 className="text-2xl font-bold text-green-800">Моята градина</h1>
          <p className="text-gray-400 text-sm">{plants.length} растения</p>
        </div>
        <button onClick={() => setShowAdd(!showAdd)} className="bg-green-600 text-white px-4 py-2 rounded-xl text-sm font-medium hover:bg-green-700">
          + Добави
        </button>
      </div>

      {showAdd && (
        <FreemiumGate show={!canAddPlant} type="plants">
          <div className="bg-white border border-green-100 rounded-2xl p-4 mb-4 shadow-sm">
            <h3 className="font-semibold text-gray-700 mb-3">Добави растение</h3>
            <select
              value={selectedId}
              onChange={e => setSelectedId(e.target.value)}
              className="w-full border border-gray-300 rounded-xl px-3 py-2.5 text-sm mb-3 focus:outline-none focus:ring-2 focus:ring-green-400"
            >
              <option value="">-- Избери растение --</option>
              {catalog.map(p => <option key={p.id} value={p.id}>{p.emoji} {p.name}</option>)}
            </select>
            <div className="grid grid-cols-2 gap-2 mb-3">
              <div>
                <label className="block text-xs text-gray-500 mb-1">Последно поливане</label>
                <input type="date" value={lastWatered} onChange={e => setLastWatered(e.target.value)}
                  className="w-full border border-gray-300 rounded-xl px-3 py-2 text-sm focus:outline-none focus:ring-2 focus:ring-green-400" />
              </div>
              <div>
                <label className="block text-xs text-gray-500 mb-1">Последно торене</label>
                <input type="date" value={lastFertilized} onChange={e => setLastFertilized(e.target.value)}
                  className="w-full border border-gray-300 rounded-xl px-3 py-2 text-sm focus:outline-none focus:ring-2 focus:ring-green-400" />
              </div>
            </div>
            <div className="flex gap-2">
              <button onClick={handleAdd} className="flex-1 bg-green-600 text-white py-2.5 rounded-xl text-sm font-medium hover:bg-green-700">Добави</button>
              <button onClick={() => setShowAdd(false)} className="flex-1 border border-gray-300 text-gray-600 py-2.5 rounded-xl text-sm hover:bg-gray-50">Откажи</button>
            </div>
          </div>
        </FreemiumGate>
      )}

      {plants.length === 0 && (
        <div className="text-center py-12 text-gray-400">
          <div className="text-5xl mb-3">🌱</div>
          <p className="font-medium">Нямате добавени растения</p>
          <p className="text-sm mt-1">Натиснете "+ Добави" за начало</p>
        </div>
      )}

      {plants.length > 0 && (
        <>
          <div className="flex rounded-xl overflow-hidden border border-gray-200 mb-4">
            <TabBtn active={activeTab === 'water'} onClick={() => setActiveTab('water')} color="blue"
              label="💧 Поливане" count={waterDue.length} />
            <TabBtn active={activeTab === 'fertilize'} onClick={() => setActiveTab('fertilize')} color="green"
              label="🌱 Торене" count={fertilizeDue.length} />
          </div>

          {activeTab === 'water' && (
            <>
              {waterDue.length > 0 && (
                <section className="mb-4">
                  <h2 className="text-xs font-semibold text-red-500 mb-2 uppercase tracking-wide">Трябва поливане ({waterDue.length})</h2>
                  <div className="grid gap-2">
                    {waterDue.map(p => (
                      <PlantRow key={p.id} plant={p} type="water" urgent
                        onAction={() => markWatered(p.id, p.watering_frequency_days)}
                        onRemove={() => removePlant(p.id)}
                        onPhotoView={setPhotoView}
                        onNote={(note) => updateNote(p.id, note)} />
                    ))}
                  </div>
                </section>
              )}
              {waterUpcoming.length > 0 && (
                <section>
                  <h2 className="text-xs font-semibold text-gray-400 mb-2 uppercase tracking-wide">Предстоящо</h2>
                  <div className="grid gap-2">
                    {waterUpcoming.map(p => (
                      <PlantRow key={p.id} plant={p} type="water"
                        onAction={() => markWatered(p.id, p.watering_frequency_days)}
                        onRemove={() => removePlant(p.id)}
                        onPhotoView={setPhotoView}
                        onNote={(note) => updateNote(p.id, note)} />
                    ))}
                  </div>
                </section>
              )}
            </>
          )}

          {activeTab === 'fertilize' && (
            <>
              {fertilizeDue.length > 0 && (
                <section className="mb-4">
                  <h2 className="text-xs font-semibold text-red-500 mb-2 uppercase tracking-wide">Трябва торене ({fertilizeDue.length})</h2>
                  <div className="grid gap-2">
                    {fertilizeDue.map(p => (
                      <PlantRow key={p.id} plant={p} type="fertilize" urgent
                        onAction={() => markFertilized(p.id, p.frequency_days)}
                        onRemove={() => removePlant(p.id)}
                        onPhotoView={setPhotoView}
                        onNote={(note) => updateNote(p.id, note)} />
                    ))}

              )}

                      <PlantRow key={p.id} plant={p} type="fertilize"
                        onAction={() => markFertilized(p.id, p.frequency_days)}

    </div>
  )
}

function TabBtn({ active, onClick, color, label, count }) {
  const colors = { blue: 'bg-blue-500', green: 'bg-green-600' }
  return (
    <button onClick={onClick}
      className={`flex-1 py-2.5 text-sm font-medium flex items-center justify-center gap-1.5 ${active ? `${colors[color]} text-white` : 'bg-white text-gray-600'}`}>
      {label}
      {count > 0 && (
        <span className={`text-xs px-1.5 py-0.5 rounded-full ${active ? 'bg-white text-gray-700' : 'bg-red-500 text-white'}`}>{count}</span>
      )}
    </button>
  )
}

function PlantRow({ plant, type, urgent, onAction, onRemove, onPhotoView, onNote }) {
```

```jsx
function toBase64(file) {
  return new Promise((resolve, reject) => {
    const MAX = 1200
    const img = new Image()
    const url = URL.createObjectURL(file)
    img.onload = () => {
      URL.revokeObjectURL(url)
      const scale = Math.min(1, MAX / Math.max(img.width, img.height))
      const canvas = document.createElement('canvas')
      canvas.width = img.width * scale
      canvas.height = img.height * scale
      canvas.getContext('2d').drawImage(img, 0, 0, canvas.width, canvas.height)
      resolve(canvas.toDataURL('image/jpeg', 0.85).split(',')[1])
    }
    img.onerror = reject
    img.src = url
  })
```

```js
import { doc, updateDoc } from 'firebase/firestore'

const VAPID_KEY = import.meta.env.VITE_FIREBASE_VAPID_KEY

export function useNotifications(uid) {
  useEffect(() => {
    if (!uid || !messaging) return

    async function requestAndSaveToken() {
      try {
        const permission = await Notification.requestPermission()
        if (permission !== 'granted') return

        const token = await getToken(messaging, { vapidKey: VAPID_KEY })
        if (token) {
          await updateDoc(doc(db, 'users', uid), { fcmToken: token })
        }
      } catch {
        // Тихо пропускаме — FCM не е критично
      }
    }

    requestAndSaveToken()

    const unsubscribe = onMessage(messaging, payload => {
      const { title, body } = payload.notification || {}
      if (title && Notification.permission === 'granted') {
        new Notification(title, { body, icon: '/icon.svg' })
      }
    })

    return unsubscribe
  }, [uid])
}
```

```js
exports.dailyWateringReminder = onSchedule(
  { schedule: '0 8 * * *', timeZone: 'Europe/Sofia', region: 'europe-west1' },
  async () => {
    const db = getFirestore()
    const today = new Date().toISOString().slice(0, 10)

    const usersSnap = await db.collection('users').where('fcmToken', '!=', null).get()

    const notifications = []

    for (const userDoc of usersSnap.docs) {
      const { fcmToken } = userDoc.data()
      const gardenSnap = await db.collection('users').doc(userDoc.id).collection('garden').get()

      const due = gardenSnap.docs.filter(d => {
        const data = d.data()
        const nextWatering = data.nextWatering?.toDate?.()?.toISOString?.().slice(0, 10)
        return nextWatering && nextWatering <= today
      })

      if (due.length > 0 && fcmToken) {
        notifications.push(
          getMessaging().send({
            token: fcmToken,
            notification: {
              title: 'GardenCare 🌿',
              body: `${due.length} растения чакат поливане днес`,
            },
          }).catch(() => {})
        )
      }
    }

    await Promise.all(notifications)
  }
)
```

```js
exports.setUserRoleFn = onCall(
  { region: 'europe-west1' },
  async (request) => {
    if (!request.auth) throw new HttpsError('unauthenticated', 'Login required')

    const callerDoc = await getFirestore().collection('users').doc(request.auth.uid).get()
    if (callerDoc.data()?.role !== 'admin') {
      throw new HttpsError('permission-denied', 'Admin only')
    }

    const { uid, role } = request.data
    if (!uid || !['free', 'premium', 'admin'].includes(role)) {
      throw new HttpsError('invalid-argument', 'Invalid uid or role')
    }

    await getFirestore().collection('users').doc(uid).update({ role })
    return { success: true }
  }
)
```

- [ ] **Step 3: Създай src/pages/Admin.jsx**

```jsx
import { useState, useEffect } from 'react'
import { useAuth } from '../hooks/useAuth'
import { useNavigate } from 'react-router-dom'
import { db } from '../services/firebase'
import { collection, getDocs, orderBy, query } from 'firebase/firestore'
import { callSetUserRole } from '../services/functions'

export default function Admin() {
  const { profile } = useAuth()
  const navigate = useNavigate()
  const [users, setUsers] = useState([])
  const [loading, setLoading] = useState(true)
  const [saving, setSaving] = useState(null)

  useEffect(() => {
    if (profile && profile.role !== 'admin') {
      navigate('/')
      return
    }
    if (profile?.role === 'admin') loadUsers()
  }, [profile])

  async function loadUsers() {
    const snap = await getDocs(query(collection(db, 'users'), orderBy('createdAt', 'desc')))
    setUsers(snap.docs.map(d => ({ id: d.id, ...d.data() })))
    setLoading(false)
  }

  async function changeRole(uid, role) {
    setSaving(uid)
    try {
      await callSetUserRole(uid, role)
      setUsers(prev => prev.map(u => u.id === uid ? { ...u, role } : u))
    } catch (e) {
      alert('Грешка: ' + e.message)
    } finally {
      setSaving(null)
    }
  }

  if (loading) return <div className="text-center py-12 text-gray-400">Зареждане...</div>

  const roleColors = { free: 'text-gray-600', premium: 'text-amber-600', admin: 'text-red-600' }
  const roleOptions = ['free', 'premium', 'admin']

  return (
    <div>
      <h1 className="text-2xl font-bold text-red-700 mb-1">Admin Panel</h1>
      <p className="text-gray-500 mb-6 text-sm">{users.length} потребители</p>

      <div className="space-y-2">
        {users.map(u => (
          <div key={u.id} className="bg-white rounded-2xl border border-gray-100 shadow-sm px-4 py-3 flex items-center gap-3">
            <div className="w-8 h-8 rounded-full bg-green-100 flex items-center justify-center text-green-700 text-sm font-bold shrink-0">
              {(u.displayName || u.email || '?')[0].toUpperCase()}
            </div>
            <div className="flex-1 min-w-0">
              <div className="text-sm font-medium text-gray-800 truncate">{u.displayName || '—'}</div>
              <div className="text-xs text-gray-400 truncate">{u.email}</div>
            </div>
            <select
              value={u.role || 'free'}
              onChange={e => changeRole(u.id, e.target.value)}
              disabled={saving === u.id}
              className={`text-xs font-semibold border border-gray-200 rounded-lg px-2 py-1 focus:outline-none focus:ring-1 focus:ring-green-400 ${roleColors[u.role || 'free']}`}
            >
              {roleOptions.map(r => (
                <option key={r} value={r}>{r}</option>
              ))}
            </select>
            {saving === u.id && <span className="text-xs text-gray-400">...</span>}
          </div>
        ))}
      </div>
    </div>
  )
}
```
