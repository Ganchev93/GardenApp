> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Мигрирай всички PoC pages към Firebase, добави Header/Footer/Profile към съществуващия layout, зони (zones-lite), дневник с реколта, weather-aware поливане, FCM push notifications, freemium gates и admin panel. Deploy в production.

Мигрираме UI-то от localStorage към Firebase hooks, добавяме FCM и Admin panel достъпен само при `role === 'admin'`.

**Tech Stack:** React 19, Vite, Tailwind CSS v4, React Router v7, Firebase v11+, lucide-react, Vitest

## Global Constraints

- **Дизайн: НЕ преизграждай визията.** Палитрата е в `src/index.css` `@theme` (forest `#1E3A2F`, sage `#4A7C59`, mint `#D4EDE0`, parchment `#EDE8DF`, amber `#C97D0E`), шрифтове Fraunces + Plus Jakarta Sans, икони lucide-react. БЕЗ емоджита в UI елементи (емоджитата на растенията 🍅 са съдържание — те остават)
- Breakpoints: mobile-first; desktop = `lg:` (≥1024px), както в текущия App.jsx — НЕ `md:`
- Sidebar на `lg:+`, BottomNav под `lg:` (всички табове с етикети), Header sticky top-0, Footer само desktop (`hidden lg:block`)
- Име на полето за торене: `fertilizing_frequency_days` (Foundation схемата)
- `useGarden(uid)` идва от Foundation Plan — не дублирай localStorage логика; `removePlant` е soft delete (`removedAt`)
- Freemium limits: free = 3 растения + 20 AI анализа/месец; premium = unlimited; admin = unlimited
- Admin panel достъпен само при `profile.role === 'admin'` — проверка на frontend И backend
- FCM: поискай permission само след user interaction, не при зареждане
- Chat: role `'error'` съобщенията НИКОГА не се пращат към API-то (филтрирай ги от историята)
- Commit след всяка завършена задача

```js
import { getToken, onMessage } from 'firebase/messaging'

const VAPID_KEY = import.meta.env.VITE_FIREBASE_VAPID_KEY

export function useNotifications(uid) {
  useEffect(() => {
    if (!uid || !messaging) return

    async function requestAndSaveToken() {
      try {
        const permission = await Notification.requestPermission()
        if (permission !== 'granted') return

        const token = await getToken(messaging, { vapidKey: VAPID_KEY })
        if (token) {
          await updateDoc(doc(db, 'users', uid), { fcmToken: token })
        }
      } catch {
        // Тихо пропускаме — FCM не е критично
      }
    }

    requestAndSaveToken()

    const unsubscribe = onMessage(messaging, payload => {
      const { title, body } = payload.notification || {}
      if (title && Notification.permission === 'granted') {
        new Notification(title, { body, icon: '/pwa-192.png' })
      }
    })

    return unsubscribe
  }, [uid])
}
```

```js
import { useState, useEffect } from 'react'
import { db } from '../services/firebase'
import { collection, onSnapshot, addDoc, deleteDoc, doc, serverTimestamp } from 'firebase/firestore'

export function useZones(uid) {
  const [zones, setZones] = useState([])
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    if (!uid) return
    return onSnapshot(collection(db, 'users', uid, 'zones'), snap => {
      setZones(snap.docs.map(d => ({ id: d.id, ...d.data() })))
      setLoading(false)
    })
  }, [uid])

  async function addZone(name) {
    const trimmed = name.trim()
    if (!trimmed) return
    await addDoc(collection(db, 'users', uid, 'zones'), { name: trimmed, createdAt: serverTimestamp() })
  }

  async function removeZone(id) {
    await deleteDoc(doc(db, 'users', uid, 'zones', id))
  }

  return { zones, addZone, removeZone, loading }
}
```

```js
import { useState, useEffect } from 'react'
import { db } from '../services/firebase'
import { collection, onSnapshot, addDoc, Timestamp } from 'firebase/firestore'

export function useHarvests(uid) {
  const [harvests, setHarvests] = useState([])
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    if (!uid) return
    return onSnapshot(collection(db, 'users', uid, 'harvests'), snap => {
      const items = snap.docs.map(d => ({ id: d.id, ...d.data() }))
      items.sort((a, b) => b.date.toMillis() - a.date.toMillis())
      setHarvests(items)
      setLoading(false)
    })
  }, [uid])

  async function addHarvest({ gardenEntryId, plantId, name, emoji, amount, unit, note = '' }) {
    await addDoc(collection(db, 'users', uid, 'harvests'), {
      gardenEntryId, plantId, name, emoji,
      amount: Number(amount), unit, note,
      date: Timestamp.fromDate(new Date()),
    })
  }

  return { harvests, addHarvest, loading }
}
```

```js
export function shouldPostponeWatering(forecast) {
  if (!forecast?.list) return { postpone: false, rainAmount: 0 }
  const next24h = forecast.list.slice(0, 8)
  const rainAmount = next24h.reduce((sum, f) => sum + (f.rain?.['3h'] || 0), 0)
  return { postpone: rainAmount > 5, rainAmount: Math.round(rainAmount) }
}
```
